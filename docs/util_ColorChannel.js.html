/**

rune.util.ColorChannel = function(value) {
    
    //--------------------------------------------------------------------------
    // Private properties
    //--------------------------------------------------------------------------
    
    /**
     * Current channel value.
     *
     * @type {number}
     * @private
     */
    this.m_value = rune.util.Math.clamp(value, 0x0, 0xff);
};

//------------------------------------------------------------------------------
// Public prototype getter and setter methods
//------------------------------------------------------------------------------

/**
 * The current value of the color channel. A channel is represented by 8 bits 
 * and can thus be a value between 0 and 255. Invalid values will be clamped 
 * within the allowable range.
 *
 * @member {number} value
 * @memberof rune.util.ColorChannel
 * @instance
 */
Object.defineProperty(rune.util.ColorChannel.prototype, "value", {
    /**
     * @this rune.util.ColorChannel
     * @ignore
     */
    get : function() {
        return this.m_value;
    },
    
    /**
     * @this rune.util.ColorChannel
     * @ignore
     */
    set : function(value) {
        this.m_value = rune.util.Math.clamp(value, 0x0, 0xff);
    }
});

/**
 * ...
 *
 * @member {number} factor
 * @memberof rune.util.ColorChannel
 * @instance
 */
Object.defineProperty(rune.util.ColorChannel.prototype, "factor", {
    /**
     * @this rune.util.ColorChannel
     * @ignore
     */
    get : function() {
        return this.m_value / 0xff;
    },
    
    /**
     * @this rune.util.ColorChannel
     * @ignore
     */
    set : function(value) {
        this.m_value = rune.util.Math.clamp(value, 0.0, 1.0) * 0xff;
    }
});
