/**
 * Creates a new color object.
 *
 * @constructor
 *
 * @class
 * @classdesc
 *
 * Represents a 32-bit color. The color consists of four 8-bit (0-255) channels;
 * red, green, blue and transparency (alpha).
 */
rune.util.Color = function() {
    
    //--------------------------------------------------------------------------
    // Private properties
    //--------------------------------------------------------------------------
    
    /**
     * Red color channel.
     *
     * @type {rune.util.ColorChannel}
     * @private
     */
    this.m_r = new rune.util.ColorChannel(0x0);
    
    /**
     * Green color channel.
     *
     * @type {rune.util.ColorChannel}
     * @private
     */
    this.m_g = new rune.util.ColorChannel(0x0);
    
    /**
     * Blue color channel.
     *
     * @type {rune.util.ColorChannel}
     * @private
     */
    this.m_b = new rune.util.ColorChannel(0x0);
    
    /**
     * Alpha channel.
     *
     * @type {rune.util.ColorChannel}
     * @private
     */
    this.m_a = new rune.util.ColorChannel(0x0);
};

    this.m_a = null;

/**
 * Returns an RGBA string representation.
 *
 * @return {string}
 */
rune.util.Color.prototype.toString = function() {
    var r = this.m_r['value'];
    var g = this.m_g['value'];
    var b = this.m_b['value'];
    var a = this.m_a['factor'];

    return "rgba(" + r + "," + g + "," + b + "," + a + ")";
};
