/**
 * ...
 *
 * @param {rune.display.Palette} [palette] ...
 * 
 * @return {undefined}
 * 
 * @private
 */
rune.display.Palettes.prototype.m_loadFlat = function(palette) {
    var canvas = this.m_context.canvas;
    var image  = this.m_context.getImageData(0, 0, canvas.width, canvas.height);
    var pixels = image.data;

    for (var i = 0; i < pixels.length; i += 4) {
        var r = pixels[i    ];
        var g = pixels[i + 1];
        var b = pixels[i + 2];
        
        var c = palette.convert(new rune.color.Color24(r,g,b));
            c = rune.color.Color24.fromHex(c);
            
        pixels[i + 0] = c['r']['value'];
        pixels[i + 1] = c['g']['value'];
        pixels[i + 2] = c['b']['value'];
    }

   this.m_context.putImageData(image, 0, 0);
}
